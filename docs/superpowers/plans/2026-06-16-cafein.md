# cafein Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a minimal macOS menu bar app (`cafein`) that prevents system sleep with timer options, live status, and an auto-disable notification.

**Architecture:** A single SwiftUI `MenuBarExtra` app (no dock icon). `SleepManager` is an `@Observable` single source of truth that holds an IOKit power assertion and a countdown timer; it depends on two injectable protocols — `PowerAssertionControlling` (IOKit wrapper) and `AutoDisableNotifying` (UserNotifications wrapper) — so its state machine is unit-testable without real system APIs. `MenuView` renders state and calls manager methods.

**Tech Stack:** Swift 5.9+, SwiftUI, `IOKit.pwr_mgt`, `UserNotifications`, `Observation`. Target: macOS 14 (Sonoma). Built with Xcode 15+.

> **Build environment:** This repo is authored on Windows but the app only compiles on macOS. All `xcodebuild` / build / run steps below must be executed on a Mac with Xcode 15+. When implementing on Windows, write the source files as specified; defer build/test verification to a macOS machine.

A tiny macOS menu bar app that prevents your Mac from sleeping, with timer options.

- [ ] **Step 1: Write the protocol and IOKit implementation**

```swift
import Foundation
import IOKit.pwr_mgt

/// Abstraction over a system power assertion so SleepManager can be tested
/// without touching IOKit.
protocol PowerAssertionControlling: AnyObject {
    var isHeld: Bool { get }
    /// Acquire a "no idle system sleep" assertion. Returns true on success
    /// (or if already held). Idempotent.
    func acquire(reason: String) -> Bool
    /// Release the assertion if held. Idempotent.
    func release()
}

/// Real implementation backed by IOKit. Uses `kIOPMAssertionTypeNoIdleSleep`,
/// which prevents system idle sleep but allows the display to sleep. No sudo.
final class IOKitPowerAssertion: PowerAssertionControlling {
    private var assertionID: IOPMAssertionID = 0
    private(set) var isHeld = false

    func acquire(reason: String) -> Bool {
        guard !isHeld else { return true }
        let result = IOPMAssertionCreateWithName(
            kIOPMAssertionTypeNoIdleSleep as CFString,
            IOPMAssertionLevel(kIOPMAssertionLevelOn),
            reason as CFString,
            &assertionID
        )
        isHeld = (result == kIOReturnSuccess)
        return isHeld
    }

    func release() {
        guard isHeld else { return }
        IOPMAssertionRelease(assertionID)
        assertionID = 0
        isHeld = false
    }
}
```

```swift
import Foundation
import UserNotifications

/// Abstraction over local notifications so SleepManager can be tested without
/// the notification center.
protocol AutoDisableNotifying: AnyObject {
    /// Request authorization lazily (safe to call repeatedly).
    func requestAuthorizationIfNeeded()
    /// Post the "sleep re-enabled" notification (timer expiry path).
    func notifyAutoDisabled()
}

final class NotificationService: AutoDisableNotifying {
    private let center = UNUserNotificationCenter.current()
    private var didRequest = false

    func requestAuthorizationIfNeeded() {
        guard !didRequest else { return }
        didRequest = true
        center.requestAuthorization(options: [.alert, .sound]) { _, _ in }
    }

    func notifyAutoDisabled() {
        let content = UNMutableNotificationContent()
        content.title = "cafein"
        content.body = "Timer ended — your Mac can sleep again."
        content.sound = .default
        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: nil // deliver immediately
        )
        center.add(request)
    }
}
```

    var acquireCount = 0
    var releaseCount = 0

    func acquire(reason: String) -> Bool {

        if acquireSucceeds { isHeld = true }
        return isHeld

    func release() {

        isHeld = false

```swift
import Foundation
import Observation

@Observable
final class SleepManager {
    enum Mode: Equatable { case manual, timed }

    private(set) var isActive = false
    private(set) var mode: Mode = .manual
    private(set) var remainingSeconds: Int?

    private let assertion: PowerAssertionControlling
    private let notifier: AutoDisableNotifying
    private let autoStartTimer: Bool
    private var timer: Timer?

    init(
        assertion: PowerAssertionControlling = IOKitPowerAssertion(),
        notifier: AutoDisableNotifying = NotificationService(),
        autoStartTimer: Bool = true
    ) {
        self.assertion = assertion
        self.notifier = notifier
        self.autoStartTimer = autoStartTimer
    }

    /// Enable sleep prevention. `duration == nil` means "until disabled".
    func enable(duration: TimeInterval?) {
        notifier.requestAuthorizationIfNeeded()
        guard assertion.acquire(reason: "cafein is keeping your Mac awake") else {
            // Acquire failed: stay inactive, don't pretend it worked.
            isActive = false
            return
        }
        invalidateTimer()
        isActive = true
        if let duration {
            mode = .timed
            remainingSeconds = Int(duration)
            if autoStartTimer { startTimer() }
        } else {
            mode = .manual
            remainingSeconds = nil
        }
    }

    /// Disable sleep prevention. `notify` true only on automatic timer expiry.
    func disable(notify: Bool = false) {
        invalidateTimer()
        assertion.release()
        isActive = false
        mode = .manual
        remainingSeconds = nil
        if notify { notifier.notifyAutoDisabled() }
    }

    /// Decrement the countdown; auto-disables (with notification) at zero.
    /// Separated from the Timer so tests can drive it deterministically.
    func advanceTimer(by seconds: Int) {
        guard mode == .timed, let remaining = remainingSeconds else { return }
        let next = remaining - seconds
        if next <= 0 {
            disable(notify: true)
        } else {
            remainingSeconds = next
        }
    }

    private func startTimer() {
        timer = Timer.scheduledTimer(withTimeInterval: 1, repeats: true) { [weak self] _ in
            self?.advanceTimer(by: 1)
        }
    }

    private func invalidateTimer() {
        timer?.invalidate()
        timer = nil
    }

    deinit {
        timer?.invalidate()
        assertion.release()
    }
}
```

import Foundation

/// Format a remaining-seconds count as "m:ss" (under an hour) or "h:mm:ss".
func formatRemaining(seconds: Int) -> String {
    let s = max(0, seconds)
    let hours = s / 3600
    let minutes = (s % 3600) / 60
    let secs = s % 60
    if hours > 0 {
        return String(format: "%d:%02d:%02d", hours, minutes, secs)
    }
    return String(format: "%d:%02d", minutes, secs)
}
```

import SwiftUI

struct MenuView: View {
    @Bindable var manager: SleepManager

    private let durations: [(label: String, seconds: TimeInterval?)] = [
        ("30 minutes", 30 * 60),
        ("1 hour", 60 * 60),
        ("2 hours", 2 * 60 * 60),
        ("Until disabled", nil)
    ]

    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            statusRow

            Divider()

            Button(action: togglePrimary) {
                Label(
                    manager.isActive ? "Disable No Sleep" : "Enable No Sleep",
                    systemImage: manager.isActive ? "moon.zzz.fill" : "cup.and.saucer.fill"
                )
                .frame(maxWidth: .infinity)
            }
            .controlSize(.large)
            .buttonStyle(.borderedProminent)

            VStack(alignment: .leading, spacing: 4) {
                Text("Keep awake for")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                ForEach(durations, id: \.label) { option in
                    Button {
                        manager.enable(duration: option.seconds)
                    } label: {
                        HStack {
                            Text(option.label)
                            Spacer()
                            if isSelected(option.seconds) {
                                Image(systemName: "checkmark")
                                    .foregroundStyle(.tint)
                            }
                        }
                        .contentShape(Rectangle())
                    }
                    .buttonStyle(.plain)
                }
            }

            Divider()

            Button("Quit cafein") {
                NSApplication.shared.terminate(nil)
            }
            .keyboardShortcut("q")
            .frame(maxWidth: .infinity, alignment: .leading)
        }
        .padding(14)
        .frame(width: 240)
    }

    private var statusRow: some View {
        HStack(spacing: 8) {
            Circle()
                .fill(manager.isActive ? Color.green : Color.secondary.opacity(0.4))
                .frame(width: 10, height: 10)
            VStack(alignment: .leading, spacing: 2) {
                Text(manager.isActive ? "No Sleep: ON" : "No Sleep: OFF")
                    .font(.headline)
                if let remaining = manager.remainingSeconds {
                    Text("Remaining \(formatRemaining(seconds: remaining))")
                        .font(.caption)
                        .foregroundStyle(.secondary)
                        .monospacedDigit()
                }
            }
            Spacer()
        }
    }

    private func isSelected(_ seconds: TimeInterval?) -> Bool {
        guard manager.isActive else { return false }
        switch (manager.mode, seconds, manager.remainingSeconds) {
        case (.manual, nil, _):
            return true
        default:
            return false
        }
    }

    private func togglePrimary() {
        if manager.isActive {
            manager.disable()
        } else {
            manager.enable(duration: nil)
        }
    }
}
```
